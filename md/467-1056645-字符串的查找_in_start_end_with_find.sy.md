### 判断函数 

```
def containsTel(s: str):
    for i in range(len(s)-10):
        if s[i] == "1":
            if s[i:i+11].isnumeric():
                return True
    return False


if __name__ == "__main__":
    s = "13901234567oeasy"
    print(s,containsTel(s))
    s = "oeasy12345678901"
    print(s,containsTel(s))
    s = "oea12345678902oeasy"
    print(s,containsTel(s))
```
