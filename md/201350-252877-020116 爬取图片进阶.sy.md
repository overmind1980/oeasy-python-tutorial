import os

    #print(list_pic)
    pic_no = 0
    for pic in list_pic:
        print(pic.get("src"))
        response3 = requests.get(pic.get("src"))
        with open(name + "/" + str(pic_no) + ".jpg","wb") as f:
             f.write(response3.content)
        pic_no += 1
    plane_no += 1
