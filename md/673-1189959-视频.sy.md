def timestamp(txt):
    s_t=list(map(int, txt.split(':')))
    return s_t[0]*60000+s_t[1]*1000+s_t[2]
#æ‰“å¼€æºæ–‡ä»¶ï¼Œå¹¶è¯»å–ä¸€äº›è§†é¢‘ä¿¡æ¯

#è®¾ç½®å¯¼å‡ºè§†é¢‘çš„å†…å®¹

    for line in f:
        l = line.strip().split(' ')
        t = list(map(timestamp, l[0].split('-')))
        txt = l[1]
        for word in l[2:]:
            txt = txt + ' '+ word
        print(t,txt)
        captions.append([t,txt])
#å¯¹æ¯ä¸€å¸§è¿›è¡ŒåŠ å·¥
