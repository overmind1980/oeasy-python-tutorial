### 字节显示

```
from random import randint
color = 0
count = -1
print("\33[0m")
while True:
    num = randint(0, 1)
    if (count == 7):
        count = -1
        if color == 0:
        	color = 7
    	else:
    		color =0 
    print("\33[" + str(color) + "m" + str(num), end="")
    count = count + 1
```

```
from random import randint
color = 40
count = -1
print("\33[0m")
while True:
    num = randint(0, 1)
    if (count == 7):
        count = -1
        color = color + 1
        if color == 48:
            color = 40
    print("\33[" + str(color) + "m" + str(num), end="")
    count = count + 1
    num = num + 1
```
