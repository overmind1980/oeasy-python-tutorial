```python
l = [[1, 2],[3, 4]]
print("[")
for i in range(len(l)):
    print("\t[",end="")
    for j in range(len(l[i])):
        if j != len(l[i])-1:
            print(str(l[i][j])+" ,",end="")
        else:
            if i != len(l)-1:
                print(str(l[i][j])+"],")
            else:
                print(str(l[i][j])+"]")
print("]")
```
