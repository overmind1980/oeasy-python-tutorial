### 王者荣耀

```
import requests
import json
import csv
import time
import random
import os
import sys
from datetime import datetime

# 设置请求头，模拟浏览器访问
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Referer': 'https://pvp.qq.com/web201605/item.shtml',
    'Accept': 'application/json, text/javascript, */*; q=0.01',
    'Accept-Encoding': 'gzip, deflate, br',
    'Accept-Language': 'zh-CN,zh;q=0.9',
    'Connection': 'keep-alive'
}

# 道具分类映射
type_mapping = {
    1: '攻击',
    2: '法术',
    3: '防御', 
    4: '移动',
    5: '打野',
    7: '游走'
}

# 配置文件路径
RAW_DATA_FILE = 'raw_item_data.json'
CLEANED_DATA_FILE = 'cleaned_item_data.json'
CSV_FILE_PATH = '王者荣耀道具数据.csv'

# 确保目录存在
def ensure_dir(file_path):
    """确保文件所在目录存在"""
    directory = os.path.dirname(file_path)
    if directory and not os.path.exists(directory):
        os.makedirs(directory)
        print(f"创建目录: {directory}")

def print_with_timestamp(message):
    """打印带时间戳的消息"""
    timestamp = datetime.now().strftime("%H:%M:%S")
    print(f"[{timestamp}] {message}")

def fetch_with_retry(url, max_retries=3):
    """带重试机制的HTTP请求"""
    retry_count = 0
    while retry_count < max_retries:
        try:
            print_with_timestamp(f"正在请求: {url}")
            response = requests.get(url, headers=headers, timeout=15)
            response.encoding = 'utf-8'
            
            if response.status_code == 200:
                print_with_timestamp(f"请求成功: {url}")
                return response.json()
            else:
                retry_count += 1
                wait_time = random.uniform(2, 5)
                print_with_timestamp(f"请求失败 (状态码: {response.status_code}), {wait_time:.1f}秒后重试... (尝试 {retry_count}/{max_retries})")
                time.sleep(wait_time)
        except Exception as e:
            retry_count += 1
            wait_time = random.uniform(2, 5)
            print_with_timestamp(f"请求异常: {e}, {wait_time:.1f}秒后重试... (尝试 {retry_count}/{max_retries})")
            time.sleep(wait_time)
    
    print_with_timestamp(f"达到最大重试次数，请求失败: {url}")
    return None

def fetch_json_data():
    """获取道具数据"""
    # 常规模式道具数据URL
    normal_items_url = 'https://pvp.qq.com/web201605/js/item.json'
    # 边境突围模式道具数据URL
    arena_items_url = 'https://pvp.qq.com/zlkdatasys/data_zlk_bjtwitem.json'
    
    all_items = []
    
    print_with_timestamp("开始获取常规模式道具数据...")
    normal_data = fetch_with_retry(normal_items_url)
    
    if normal_data:
        print_with_timestamp(f"成功获取到 {len(normal_data)} 个常规模式道具")
        # 转换常规模式道具数据格式
        for item in normal_data:
            try:
                item_data = {
                    'item_id': item.get('item_id'),
                    'item_name': item.get('item_name'),
                    'price': item.get('price'),
                    'total_price': item.get('total_price'),
                    'des1': item.get('des1'),  # 基本属性
                    'des2': item.get('des2'),  # 附加技能
                    'item_type': item.get('item_type'),
                    'category': type_mapping.get(item.get('item_type'), '未知'),
                    'mode': '常规模式'
                }
                all_items.append(item_data)
            except Exception as e:
                print_with_timestamp(f"解析常规模式道具时出错: {e}")
    else:
        print_with_timestamp("警告: 未能获取常规模式道具数据")
    
    # 添加随机延迟，避免请求过于频繁
    time.sleep(random.uniform(1, 3))
    
    print_with_timestamp("开始获取边境突围模式道具数据...")
    arena_data = fetch_with_retry(arena_items_url)
    
    if arena_data and 'bjtwzbsy_ba' in arena_data:
        arena_items = arena_data['bjtwzbsy_ba']
        print_with_timestamp(f"成功获取到 {len(arena_items)} 个边境突围模式道具")
        
        # 转换边境突围模式道具数据格式
        for item in arena_items:
            try:
                item_data = {
                    'item_id': int(item.get('itemidzbid_4a', 0)),
                    'item_name': item.get('itemnamezwm_cd', ''),
                    'price': item.get('itemlvzbdj_96', 0),
                    'total_price': None,  # 边境突围模式没有总价
                    'des1': item.get('des1zbsx_a6'),  # 基本属性
                    'des2': item.get('des2fszx_cc'),  # 附加技能
                    'item_type': int(item.get('itemtypezbfl_30', 0)),
                    'category': type_mapping.get(int(item.get('itemtypezbfl_30', 0)), '未知'),
                    'mode': '边境突围模式'
                }
                all_items.append(item_data)
            except Exception as e:
                print_with_timestamp(f"解析边境突围模式道具时出错: {e}")
    else:
        print_with_timestamp("警告: 未能获取边境突围模式道具数据")
    
    # 保存原始数据到文件
    if all_items:
        ensure_dir(RAW_DATA_FILE)
        with open(RAW_DATA_FILE, 'w', encoding='utf-8') as f:
            json.dump(all_items, f, ensure_ascii=False, indent=2)
        print_with_timestamp(f"所有道具数据已保存到 {RAW_DATA_FILE}，共 {len(all_items)} 个道具")
    
    return all_items

def clean_data(items):
    """清理和标准化数据"""
    print_with_timestamp("开始清理数据...")
    cleaned_items = []
    
    for index, item in enumerate(items):
        try:
            # 清理基本属性和附加技能文本，处理可能的None值
            des1 = str(item.get('des1', '')).strip()
            des2 = str(item.get('des2', '')).strip()
            
            # 确保价格数据类型正确
            price = item.get('price', 0)
            if isinstance(price, str):
                try:
                    price = int(price)
                except ValueError:
                    price = 0
            
            total_price = item.get('total_price')
            if total_price is not None and isinstance(total_price, str):
                try:
                    total_price = int(total_price)
                except ValueError:
                    total_price = 0
            
            cleaned_item = {
                'item_id': item.get('item_id'),
                'item_name': item.get('item_name', '').strip(),
                'price': price,
                'total_price': total_price,
                'skills': des1,
                'additional_skills': des2 if des2 else '无',
                'category': item.get('category', '未知'),
                'mode': item.get('mode', '未知')
            }
            
            cleaned_items.append(cleaned_item)
            
            # 显示进度
            if (index + 1) % 20 == 0 or index + 1 == len(items):
                print_with_timestamp(f"已处理 {index + 1}/{len(items)} 个道具")
                
        except Exception as e:
            print_with_timestamp(f"清理道具数据时出错 (索引 {index}): {e}")
    
    # 保存清理后的数据
    ensure_dir(CLEANED_DATA_FILE)
    with open(CLEANED_DATA_FILE, 'w', encoding='utf-8') as f:
        json.dump(cleaned_items, f, ensure_ascii=False, indent=2)
    
    print_with_timestamp(f"数据清理完成！共 {len(cleaned_items)} 个道具")
    print_with_timestamp(f"清理后的数据已保存到 {CLEANED_DATA_FILE}")
    
    return cleaned_items

def export_to_csv(items):
    """将道具数据导出为CSV文件"""
    print_with_timestamp("开始导出CSV文件...")
    
    # 定义CSV表头，根据用户需求包含以下字段：
    # 道具名字、售价、总价、技能、附加技能、分类
    headers = ['道具名称', '售价', '总价', '技能', '附加技能', '分类', '模式']
    
    # 确保输出目录存在
    ensure_dir(CSV_FILE_PATH)
    
    # 写入CSV文件，使用utf-8-sig编码以支持Excel
    try:
        with open(CSV_FILE_PATH, 'w', newline='', encoding='utf-8-sig') as csvfile:
            writer = csv.DictWriter(csvfile, fieldnames=headers)
            
            # 写入表头
            writer.writeheader()
            
            # 写入数据行
            for index, item in enumerate(items):
                try:
                    # 准备行数据，映射到CSV的列名
                    row = {
                        '道具名称': item.get('item_name', ''),
                        '售价': item.get('price', 0),
                        '总价': item.get('total_price', '') if item.get('total_price') is not None else '',
                        '技能': item.get('skills', '无'),
                        '附加技能': item.get('additional_skills', '无'),
                        '分类': item.get('category', '未知'),
                        '模式': item.get('mode', '未知')
                    }
                    writer.writerow(row)
                    
                    # 显示进度
                    if (index + 1) % 50 == 0 or index + 1 == len(items):
                        print_with_timestamp(f"已写入 {index + 1}/{len(items)} 条数据到CSV")
                        
                except Exception as e:
                    print_with_timestamp(f"写入CSV行时出错 (索引 {index}): {e}")
        
        print_with_timestamp(f"CSV文件导出成功！文件路径: {os.path.abspath(CSV_FILE_PATH)}")
        print_with_timestamp(f"共导出 {len(items)} 个道具数据")
        
        # 按分类统计道具数量
        print("\n按分类统计:")
        category_count = {}
        for item in items:
            category = item.get('category', '未知')
            category_count[category] = category_count.get(category, 0) + 1
        
        for category, count in sorted(category_count.items()):
            print(f"{category}: {count} 个")
        
        # 按模式统计道具数量
        print("\n按模式统计:")
        mode_count = {}
        for item in items:
            mode = item.get('mode', '未知')
            mode_count[mode] = mode_count.get(mode, 0) + 1
        
        for mode, count in sorted(mode_count.items()):
            print(f"{mode}: {count} 个")
            
        return True
        
    except Exception as e:
        print_with_timestamp(f"导出CSV时出错: {e}")
        import traceback
        traceback.print_exc()
        return False

def run_entire_process():
    """运行整个爬虫流程"""
    print("=" * 60)
    print("        王者荣耀道具数据爬虫")
    print("=" * 60)
    print_with_timestamp("爬虫开始运行...")
    
    try:
        # 步骤1: 获取数据
        items = fetch_json_data()
        
        if not items:
            print_with_timestamp("错误: 未能获取到任何道具数据")
            # 尝试从已保存的文件加载数据
            if os.path.exists(RAW_DATA_FILE):
                print_with_timestamp(f"尝试从 {RAW_DATA_FILE} 加载数据...")
                try:
                    with open(RAW_DATA_FILE, 'r', encoding='utf-8') as f:
                        items = json.load(f)
                    print_with_timestamp(f"成功从文件加载 {len(items)} 个道具数据")
                except Exception as e:
                    print_with_timestamp(f"加载文件时出错: {e}")
                    return False
            else:
                return False
        
        # 步骤2: 清理数据
        cleaned_items = clean_data(items)
        
        # 步骤3: 导出CSV
        success = export_to_csv(cleaned_items)
        
        if success:
            print("\n" + "=" * 60)
            print("爬虫任务完成！")
            print(f"CSV文件位置: {os.path.abspath(CSV_FILE_PATH)}")
            print("您可以在Excel或其他表格软件中打开此文件查看数据")
            print("=" * 60)
            return True
        else:
            return False
            
    except KeyboardInterrupt:
        print("\n爬虫被用户中断")
        return False
    except Exception as e:
        print_with_timestamp(f"爬虫运行时出错: {e}")
        import traceback
        traceback.print_exc()
        return False

def main():
    """主函数"""
    # 检查Python版本
    if sys.version_info < (3, 6):
        print("错误: 需要Python 3.6或更高版本")
        return 1
    
    # 运行爬虫流程
    success = run_entire_process()
    
    # 暂停显示，让用户有时间查看结果
    input("\n按回车键退出...")
    
    return 0 if success else 1

if __name__ == "__main__":
    sys.exit(main())
```
